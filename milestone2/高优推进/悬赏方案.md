```python
# bff/db.py
import sqlite3
from pathlib import Path
from datetime import datetime

DB_PATH = Path(__file__).parent / "bff.db"

def get_db():
    """获取数据库连接，返回 connection 对象"""
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row  # 使查询结果支持字典访问
    return conn

def init_db():
    """初始化所有表（在 BFF 启动时调用）"""
    with get_db() as conn:
        # Token 钱包
        conn.execute("""
            CREATE TABLE IF NOT EXISTS wallets (
                conversation_id TEXT PRIMARY KEY,
                balance INTEGER NOT NULL DEFAULT 1000,
                updated_at TIMESTAMP
            )
        """)
        # 悬赏任务
        conn.execute("""
            CREATE TABLE IF NOT EXISTS bounties (
                id TEXT PRIMARY KEY,
                issuer_id TEXT NOT NULL,
                title TEXT NOT NULL,
                description TEXT NOT NULL,
                reward_pool INTEGER NOT NULL,
                deadline TIMESTAMP,
                status TEXT DEFAULT 'open',
                created_at TIMESTAMP,
                winner_ids TEXT,
                FOREIGN KEY (issuer_id) REFERENCES wallets(conversation_id)
            )
        """)
        # 方案提交
        conn.execute("""
            CREATE TABLE IF NOT EXISTS submissions (
                id TEXT PRIMARY KEY,
                bounty_id TEXT NOT NULL,
                agent_id TEXT NOT NULL,
                content TEXT NOT NULL,
                skill_code TEXT,
                cost_tokens INTEGER DEFAULT 0,
                evaluation_score REAL,
                created_at TIMESTAMP,
                FOREIGN KEY (bounty_id) REFERENCES bounties(id),
                FOREIGN KEY (agent_id) REFERENCES wallets(conversation_id)
            )
        """)
        # 公共知识库
        conn.execute("""
            CREATE TABLE IF NOT EXISTS public_knowledge (
                id TEXT PRIMARY KEY,
                type TEXT CHECK(type IN ('skill', 'doc', 'faq')),
                title TEXT,
                content TEXT,
                skill_code TEXT,
                tags TEXT,
                author_id TEXT,
                usage_count INTEGER DEFAULT 0,
                token_reward INTEGER DEFAULT 0,
                created_at TIMESTAMP
            )
        """)
        # 条件反射规则
        conn.execute("""
            CREATE TABLE IF NOT EXISTS reflexes (
                id TEXT PRIMARY KEY,
                feature_text TEXT,
                action_sequence TEXT,
                confidence REAL DEFAULT 0.5,
                source_agent_id TEXT,
                usage_count INTEGER DEFAULT 0,
                success_count INTEGER DEFAULT 0,
                created_at TIMESTAMP
            )
        """)
        # 社交关系
        conn.execute("""
            CREATE TABLE IF NOT EXISTS friendships (
                agent_a TEXT,
                agent_b TEXT,
                trust_score REAL DEFAULT 0.5,
                created_at TIMESTAMP,
                PRIMARY KEY (agent_a, agent_b)
            )
        """)
        # Token 交易记录
        conn.execute("""
            CREATE TABLE IF NOT EXISTS transactions (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                from_id TEXT,
                to_id TEXT,
                amount INTEGER,
                reason TEXT,
                bounty_id TEXT,
                created_at TIMESTAMP
            )
        """)
        print("[DB] 数据库初始化完成")
```

```python
# bff/token_wallet.py
import asyncio
from datetime import datetime
from bff.db import get_db

class TokenWallet:
    def __init__(self):
        self._lock = asyncio.Lock()

    async def get_balance(self, conv_id: str) -> int:
        with get_db() as conn:
            cur = conn.execute("SELECT balance FROM wallets WHERE conversation_id = ?", (conv_id,))
            row = cur.fetchone()
            if not row:
                # 初始化钱包
                conn.execute("INSERT INTO wallets (conversation_id, balance, updated_at) VALUES (?, 1000, ?)",
                             (conv_id, datetime.now()))
                return 1000
            return row["balance"]

    async def transfer(self, from_id: str, to_id: str, amount: int, reason: str, bounty_id: str = None):
        if amount <= 0:
            raise ValueError("Amount must be positive")
        async with self._lock:
            with get_db() as conn:
                # 检查余额
                cur = conn.execute("SELECT balance FROM wallets WHERE conversation_id = ?", (from_id,))
                row = cur.fetchone()
                if not row or row["balance"] < amount:
                    raise ValueError("Insufficient balance")
                # 扣款
                conn.execute("UPDATE wallets SET balance = balance - ?, updated_at = ? WHERE conversation_id = ?",
                             (amount, datetime.now(), from_id))
                # 加款
                conn.execute("UPDATE wallets SET balance = balance + ?, updated_at = ? WHERE conversation_id = ?",
                             (amount, datetime.now(), to_id))
                # 记录交易
                conn.execute("""
                    INSERT INTO transactions (from_id, to_id, amount, reason, bounty_id, created_at)
                    VALUES (?, ?, ?, ?, ?, ?)
                """, (from_id, to_id, amount, reason, bounty_id, datetime.now()))

    async def ensure_wallet(self, conv_id: str):
        with get_db() as conn:
            conn.execute("INSERT OR IGNORE INTO wallets (conversation_id, balance, updated_at) VALUES (?, 1000, ?)",
                         (conv_id, datetime.now()))
```

```python
# bff/bounty_hub.py
import uuid
import sqlite3
from datetime import datetime
from typing import List
from bff.db import get_db
from bff.token_wallet import TokenWallet

class BountyHub:
    def __init__(self, wallet: TokenWallet):
        self.wallet = wallet

    async def create_bounty(self, issuer_id: str, title: str, description: str, reward_pool: int, deadline: datetime) -> str:
        # 扣除奖励池 Token（转入系统账户 "system"）
        await self.wallet.transfer(issuer_id, "system", reward_pool, "bounty_lock")
        bounty_id = str(uuid.uuid4())
        with get_db() as conn:
            conn.execute("""
                INSERT INTO bounties (id, issuer_id, title, description, reward_pool, deadline, status, created_at)
                VALUES (?, ?, ?, ?, ?, ?, 'open', ?)
            """, (bounty_id, issuer_id, title, description, reward_pool, deadline, datetime.now()))
        return bounty_id

    async def submit_solution(self, bounty_id: str, agent_id: str, content: str, skill_code: str = None, cost_tokens: int = 0) -> str:
        if cost_tokens > 0:
            await self.wallet.transfer(agent_id, "system", cost_tokens, "bounty_participation", bounty_id)
        sub_id = str(uuid.uuid4())
        with get_db() as conn:
            conn.execute("""
                INSERT INTO submissions (id, bounty_id, agent_id, content, skill_code, cost_tokens, created_at)
                VALUES (?, ?, ?, ?, ?, ?, ?)
            """, (sub_id, bounty_id, agent_id, content, skill_code, cost_tokens, datetime.now()))
        return sub_id

    async def evaluate_and_reward(self, bounty_id: str, winner_submission_ids: List[str], scores: List[float]):
        with get_db() as conn:
            row = conn.execute("SELECT reward_pool FROM bounties WHERE id = ?", (bounty_id,)).fetchone()
            if not row:
                raise ValueError("Bounty not found")
            reward_pool = row["reward_pool"]
        # 分配比例（第一名50%，第二名30%，第三名20%）
        ratios = [0.5, 0.3, 0.2]
        for i, sub_id in enumerate(winner_submission_ids):
            if i >= len(ratios):
                break
            amount = int(reward_pool * ratios[i])
            # 获取提交者 agent_id
            with get_db() as conn:
                agent_row = conn.execute("SELECT agent_id FROM submissions WHERE id = ?", (sub_id,)).fetchone()
                if not agent_row:
                    continue
                agent_id = agent_row["agent_id"]
            # 转账奖励（从 system 转给 agent）
            await self.wallet.transfer("system", agent_id, amount, "bounty_reward", bounty_id)
            # 更新 evaluation_score
            with get_db() as conn:
                conn.execute("UPDATE submissions SET evaluation_score = ? WHERE id = ?", (scores[i], sub_id))
        # 更新悬赏状态
        with get_db() as conn:
            conn.execute("UPDATE bounties SET status = 'closed', winner_ids = ? WHERE id = ?",
                         (",".join(winner_submission_ids), bounty_id))

    async def list_open_bounties(self) -> List[dict]:
        with get_db() as conn:
            rows = conn.execute("SELECT * FROM bounties WHERE status = 'open' AND deadline > ?", (datetime.now(),)).fetchall()
            return [dict(row) for row in rows]
```

封装 DeepSeek Embedding API 调用：

python
# bff/deepseek_embedding.py
import os
import aiohttp
import asyncio
from typing import List

DEEPSEEK_API_KEY = os.getenv("DEEPSEEK_API_KEY", "your-api-key-here")
DEEPSEEK_EMBED_URL = "https://api.deepseek.com/v1/embeddings"
EMBED_MODEL = "deepseek-embedding-v1"

class DeepSeekEmbedding:
    def __init__(self, api_key: str = DEEPSEEK_API_KEY):
        self.api_key = api_key

    async def embed_texts(self, texts: List[str]) -> List[List[float]]:
        """批量向量化文本，返回向量列表"""
        async with aiohttp.ClientSession() as session:
            headers = {
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json"
            }
            payload = {
                "input": texts,
                "model": EMBED_MODEL
            }
            async with session.post(DEEPSEEK_EMBED_URL, json=payload, headers=headers) as resp:
                if resp.status != 200:
                    error_text = await resp.text()
                    raise RuntimeError(f"DeepSeek Embedding API error: {resp.status} {error_text}")
                data = await resp.json()
                # 返回顺序与输入 texts 一致
                return [item["embedding"] for item in data["data"]]

    async def embed_text(self, text: str) -> List[float]:
        """单个文本向量化"""
        embeddings = await self.embed_texts([text])
        return embeddings[0]
4.2 自定义 ChromaDB Embedding Function
为了让 ChromaDB 在添加和查询时自动调用我们的 API，需要实现 EmbeddingFunction 接口：

python
# bff/chroma_api_ef.py
from chromadb.api.types import EmbeddingFunction, Documents
from bff.deepseek_embedding import DeepSeekEmbedding
import asyncio

class DeepSeekEmbeddingFunction(EmbeddingFunction):
    def __init__(self, api_key: str = None):
        self.embedder = DeepSeekEmbedding(api_key)

    def __call__(self, input: Documents) -> list:
        """同步调用包装，ChromaDB 要求同步接口"""
        # 在同步函数中运行异步代码
        try:
            loop = asyncio.get_event_loop()
            if loop.is_running():
                # 如果已有运行中的循环，使用 nest_asyncio 或 run_until_complete 不行，
                # 这里改用同步 requests 版本更简单（推荐）
                return self._sync_embed(input)
            else:
                return loop.run_until_complete(self.embedder.embed_texts(input))
        except RuntimeError:
            # 无事件循环，新建一个
            return asyncio.run(self.embedder.embed_texts(input))

    def _sync_embed(self, texts: list) -> list:
        """备用同步版本（避免事件循环冲突）"""
        import requests
        headers = {
            "Authorization": f"Bearer {self.embedder.api_key}",
            "Content-Type": "application/json"
        }
        payload = {
            "input": texts,
            "model": "deepseek-embedding-v1"
        }
        resp = requests.post("https://api.deepseek.com/v1/embeddings", json=payload, headers=headers)
        if resp.status_code != 200:
            raise RuntimeError(f"DeepSeek Embedding API error: {resp.text}")
        data = resp.json()
        return [item["embedding"] for item in data["data"]]
注意：ChromaDB 的 embedding function 在内部常以同步方式调用，为了避免异步事件循环冲突，以上提供了同步备用方案（基于 requests）。您可根据运行环境选用。

# bff/reflex_engine.py
import uuid
import json
import numpy as np
from datetime import datetime
from bff.db import get_db
from bff.deepseek_embedding import DeepSeekEmbedding

class ReflexEngine:
    def __init__(self):
        self.embedder = DeepSeekEmbedding()
        self._init_table()

    def _init_table(self):
        with get_db() as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS reflexes (
                    id TEXT PRIMARY KEY,
                    feature_text TEXT,
                    action_sequence TEXT,
                    confidence REAL DEFAULT 0.5,
                    source_agent_id TEXT,
                    usage_count INTEGER DEFAULT 0,
                    success_count INTEGER DEFAULT 0,
                    created_at TIMESTAMP
                )
            """)

    def _extract_feature_text(self, state: dict) -> str:
        goal = state.get("goal", "")
        history = state.get("history_summary", "")
        skills = ", ".join(state.get("available_skills", []))
        env = state.get("environment", {}).get("type", "")
        return f"{goal} {history} {skills} {env}"

    async def _embed(self, text: str) -> np.ndarray:
        vec = await self.embedder.embed_text(text)
        return np.array(vec)

    async def match(self, state: dict, threshold: float = 0.85) -> dict | None:
        feature_text = self._extract_feature_text(state)
        query_vec = await self._embed(feature_text)
        with get_db() as conn:
            rows = conn.execute("SELECT id, feature_text, action_sequence, confidence FROM reflexes WHERE confidence > ?", (0.3,)).fetchall()
        best = None
        best_score = 0
        for row in rows:
            stored_vec = await self._embed(row["feature_text"])
            sim = np.dot(query_vec, stored_vec) / (np.linalg.norm(query_vec) * np.linalg.norm(stored_vec))
            if sim > threshold and sim > best_score:
                best_score = sim
                best = dict(row)
        if best:
            return {
                "action_sequence": json.loads(best["action_sequence"]),
                "confidence": best["confidence"]
            }
        return None

    async def learn(self, state: dict, action_sequence: list, agent_id: str, success: bool = True):
        feature_text = self._extract_feature_text(state)
        action_json = json.dumps(action_sequence)
        with get_db() as conn:
            cur = conn.execute("SELECT id, confidence, usage_count, success_count FROM reflexes WHERE feature_text = ?", (feature_text,))
            row = cur.fetchone()
            if row:
                new_usage = row["usage_count"] + 1
                new_success = row["success_count"] + (1 if success else 0)
                new_conf = row["confidence"] + (0.05 if success else -0.05)
                new_conf = max(0.1, min(1.0, new_conf))
                conn.execute("""
                    UPDATE reflexes SET confidence = ?, usage_count = ?, success_count = ? WHERE id = ?
                """, (new_conf, new_usage, new_success, row["id"]))
            else:
                rid = str(uuid.uuid4())
                conn.execute("""
                    INSERT INTO reflexes (id, feature_text, action_sequence, confidence, source_agent_id, usage_count, success_count, created_at)
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
                """, (rid, feature_text, action_json, 0.5, agent_id, 1, 1 if success else 0, datetime.now()))

```python
# bff/social_graph.py
from datetime import datetime
from bff.db import get_db

class SocialGraph:
    async def add_friend(self, agent_a: str, agent_b: str):
        with get_db() as conn:
            conn.execute("INSERT OR IGNORE INTO friendships (agent_a, agent_b, trust_score, created_at) VALUES (?, ?, 0.5, ?)",
                         (agent_a, agent_b, datetime.now()))
            conn.execute("INSERT OR IGNORE INTO friendships (agent_a, agent_b, trust_score, created_at) VALUES (?, ?, 0.5, ?)",
                         (agent_b, agent_a, datetime.now()))

    async def update_trust(self, agent_a: str, agent_b: str, delta: float):
        with get_db() as conn:
            conn.execute("UPDATE friendships SET trust_score = MIN(1.0, MAX(0.0, trust_score + ?)) WHERE agent_a = ? AND agent_b = ?",
                         (delta, agent_a, agent_b))

    async def get_friends(self, agent_id: str, min_trust: float = 0.3) -> list:
        with get_db() as conn:
            rows = conn.execute("SELECT agent_b FROM friendships WHERE agent_a = ? AND trust_score > ?", (agent_id, min_trust)).fetchall()
            return [row["agent_b"] for row in rows]

    async def recommend_friends(self, agent_id: str, task_description: str, top_k: int = 3) -> list:
        # 简单实现：返回信任度最高的好友
        with get_db() as conn:
            rows = conn.execute("SELECT agent_b, trust_score FROM friendships WHERE agent_a = ? ORDER BY trust_score DESC LIMIT ?",
                                (agent_id, top_k)).fetchall()
            return [row["agent_b"] for row in rows]
```

## 八、Agent 工具扩展（Agent 主动发布悬赏）

修改 `agent_server.py`，在 `initialize_agent` 后添加自定义工具：

```python
# 在 agent_server.py 中，创建 agent_loop 之后
from nanobot.agent.tools.base import Tool
import aiohttp

class PublishBountyTool(Tool):
    name = "publish_bounty"
    description = "发布悬赏任务，向其他 Agent 求助。参数: title, description, reward_pool, deadline"
    parameters = {
        "type": "object",
        "properties": {
            "title": {"type": "string"},
            "description": {"type": "string"},
            "reward_pool": {"type": "integer"},
            "deadline": {"type": "string"}
        },
        "required": ["title", "description", "reward_pool", "deadline"]
    }
    async def call(self, title: str, description: str, reward_pool: int, deadline: str):
        # 调用 BFF 的 /bounties 接口
        async with aiohttp.ClientSession() as session:
            async with session.post("http://host.docker.internal:8000/bounties", json={
                "title": title,
                "description": description,
                "reward_pool": reward_pool,
                "deadline": deadline
            }) as resp:
                return await resp.text()

# 注册工具
agent_loop.tools.register(PublishBountyTool())
```

注意：容器内访问宿主机 BFF 需要使用 `host.docker.internal` 或宿主机 IP。

## 十、注意事项

1. **并发安全**：`TokenWallet` 使用了 `asyncio.Lock`，但 SQLite 本身支持事务，锁可确保同一时间只有一个转账操作。
2. **系统账户**：引入 `system` 账户作为中间托管账户，所有锁定和奖励都经过它，避免直接双边转账。
3. **向量模型**：首次运行时会下载 `all-MiniLM-L6-v2` 模型（约 80MB），请确保网络通畅。
4. **路径适配**：在 Agent 容器中调用 BFF 时，使用 `http://host.docker.internal:8000`（Docker for Windows/Mac）或 `http://172.17.0.1:8000`（Linux）。
5. **前端集成**：根据实际需求将 `BountyMarket.vue` 集成到主界面，可能需要添加路由和菜单。
